\begin{solution}
  $\displaystyle \int_0^\infty e^{-x} dx = 1$
  \begin{theorem}[Pythagorean Theorem]
    \label{pyth}
    If $x$, $y$, and $z$ are the lengths of the legs of a right triangle, then
    \[
      x^2 + y^2 = z^2
    \]
  \end{theorem}

  I am referencing~\ref{pyth}.
\end{solution}
